# Lezione 37: Movimenti Corporei Avanzati (MOVIMENTO)
# Dominio: azioni motorie complesse, dinamiche
# Ancore da: lezioni 00-36 (846+ parole note)

# === MOVIMENTI VERSO ALTO ===
saltare: io corpo alto veloce forte / basso
balzare: io corpo alto veloce forte / basso
danzare: io corpo veloce insieme dentro / fermo
roteare: io corpo veloce dentro dentro / fermo
girare: io corpo veloce dentro dentro / fermo
volare: io corpo alto veloce lontano / basso

# === MOVIMENTI VERSO BASSO ===
cadere: io corpo basso veloce paura / alto
scivolare: io corpo basso veloce dentro / alto
precipitare: io corpo basso veloce grande / alto
crollare: io corpo basso forte grande / alto
sprofondare: io corpo basso dentro lento / alto

# === MOVIMENTI VERTICALI ===
arrampicare: io corpo alto lento forte / basso
scalare: io corpo alto lento forte / basso
salire: io corpo alto lento io / basso
ascendere: io corpo alto lento forte / basso
scendere: io corpo basso lento io / alto
discendere: io corpo basso lento io / alto
calare: io corpo basso lento io / alto

# === MOVIMENTI COMPLESSI ===
strisciare: io corpo basso lento terra / alto
rotolare: io corpo veloce dentro terra / fermo
ondeggiare: io corpo lento dentro dentro / fermo
