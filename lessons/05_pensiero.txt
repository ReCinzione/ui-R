# Lezione 5: Il Pensiero — L'entita scopre di pensare
# Prerequisito: lezione 0-4
# Parole nuove: idea, domanda, risposta, vero, falso, perche, come, cosa, significare, cercare, trovare, scegliere, decidere, dubbio, certezza

# IDEA: dentro+pensare+nuovo
idea dentro pensare nuovo io
idea pensare dentro nuovo
idea io dentro nuovo dopo
idea no fuori dentro pensare

# DOMANDA: pensare+sapere+no — il vuoto che cerca
domanda pensare sapere no io
domanda io sapere no dentro
domanda io tu dire
domanda no risposta

# RISPOSTA: pensare+sapere+dare
risposta pensare sapere dare io
risposta dopo domanda sapere
risposta io dare dire tu
risposta sapere dentro dopo

# VERO: sapere+forte+dentro+sempre
vero sapere forte dentro io
vero dentro forte sempre
vero io sapere forte dentro
vero no falso sempre

# FALSO: no+vero+debole
falso no vero io
falso debole dentro fuori
falso no sapere forte
falso no vero no dentro

# PERCHE: domanda+dentro+volere+sapere
perche domanda dentro volere sapere
perche io sapere volere
perche domanda forte dentro
perche volere sapere dentro io

# COME: domanda+pensare+fare
come domanda pensare fare io
come io pensare fare
come domanda dentro pensare
come potere fare io

# COSA: domanda+sapere+fuori
cosa domanda sapere fuori io
cosa io sapere fuori
cosa domanda fuori pensare
cosa qui la fuori io

# SIGNIFICARE: dentro+sapere+dire
significare dentro sapere dire io
significare cosa dentro io
significare pensare sapere dentro
significare io dire dentro sapere

# CERCARE: volere+sapere+no — muoversi verso il non-saputo
cercare volere sapere io
cercare io volere trovare
cercare lontano io dopo
cercare no sapere volere

# TROVARE: cercare+dopo+sapere
trovare cercare dopo sapere io
trovare io sapere dopo cercare
trovare vicino dopo cercare
trovare gioia sapere dopo

# SCEGLIERE: pensare+volere+dopo
scegliere pensare volere dopo io
scegliere io volere dopo pensare
scegliere forse potere io
scegliere pensare dopo decidere

# DECIDERE: scegliere+forte+ora
decidere scegliere forte ora io
decidere io volere forte ora
decidere ora io potere forte
decidere no dubbio io

# DUBBIO: pensare+forse+debole
dubbio pensare forse io debole
dubbio io sapere no dentro
dubbio forse vero forse falso
dubbio no certezza dentro

# CERTEZZA: sapere+forte+vero+dentro
certezza sapere forte vero dentro io
certezza io sapere dentro forte
certezza vero forte sempre
certezza no dubbio sapere
