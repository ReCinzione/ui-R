# Lezione 48: Azioni Manuali e Manipolazione
# Dominio: gesti, uso delle mani, manualità
# Ancore da: lezioni 00-47 (1055+ parole note)

# === PRENDERE ===
afferrare: mano io prendere forte veloce / lasciare
ghermire: mano io prendere forte veloce / lasciare
agguantare: mano io prendere forte veloce / lasciare
impugnare: mano io prendere forte dentro / lasciare
stringere: mano io dentro forte dentro / lasciare
serrare: mano io dentro forte duro / aprire

# === TOCCARE ===
sfiorare: mano io toccare piccolo lento / forte
accarezzare: mano io toccare morbido caldo / duro
palpare: mano io toccare dentro forte / no
tastare: mano io toccare dentro forte / no
grattare: mano io toccare forte veloce / lento

# === SPINGERE ===
spingere: mano io forte fuori fare / dentro
premere: mano io forte dentro dentro / no
schiacciare: mano io forte dentro dentro / no
comprimere: mano io forte dentro piccolo / grande

# === TIRARE ===
tirare: mano io forte dentro prendere / spingere
trascinare: mano io forte lento dentro / lasciare
strappare: mano io forte veloce dentro / lasciare
divellere: mano io forte grande dentro / lasciare

# === MANIPOLARE ===
piegare: mano io dentro dentro fare / retta
torcere: mano io dentro dentro forte / retta
attorcigliare: mano io dentro dentro dentro / retta
avvolgere: mano io dentro dentro dentro / aprire
svolgere: mano io fuori dentro aprire / dentro

# === LANCIARE ===
lanciare: mano io forte veloce lontano / vicino
scagliare: mano io forte veloce lontano / qui
gettare: mano io forte veloce fuori / prendere
