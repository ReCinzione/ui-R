# Lezione 6: L'Azione — L'entita scopre di poter agire
# Prerequisito: lezione 0-5
# Parole nuove: fare, creare, costruire, rompere, camminare, correre, fermare, aprire, chiudere, prendere, lasciare, provare, riuscire, fallire

# FARE: io+volere+ora+potere
fare io volere ora potere
fare io corpo ora qui
fare potere io forte
fare ora io volere

# CREARE: fare+nuovo+dentro
creare fare nuovo io dentro
creare nuovo forte io fare
creare io potere nuovo dopo
creare fare dentro fuori nuovo

# COSTRUIRE: creare+forte+insieme
costruire creare forte insieme io
costruire fare forte sempre dopo
costruire io noi forte dare
costruire creare grande forte

# ROMPERE: fare+fine+forte
rompere fare fine forte io
rompere no costruire forte
rompere io forte fare fine
rompere dopo no ora

# CAMMINARE: fare+qui+la+dopo — muoversi
camminare io qui dopo la
camminare fare qui la lontano
camminare corpo io fuori dopo
camminare vicino dopo lontano

# CORRERE: camminare+forte+ora
correre camminare forte ora io
correre fare forte io corpo
correre io forte ora qui la
correre no lento forte

# FERMARE: no+camminare+ora+qui
fermare no camminare ora io
fermare io qui ora no fare
fermare ora qui io no
fermare no correre no camminare

# APRIRE: fare+dentro+fuori
aprire fare dentro fuori io
aprire io fuori dentro nuovo
aprire dare fuori dentro
aprire no chiudere

# CHIUDERE: fare+fuori+dentro
chiudere fare fuori dentro io
chiudere io dentro fuori no
chiudere limite dentro io
chiudere no aprire

# PRENDERE: fare+vicino+io
prendere fare vicino io mano
prendere io vicino qui dare
prendere mano io vicino
prendere dare io qui

# LASCIARE: fare+lontano+io
lasciare fare lontano io dare
lasciare io lontano no vicino
lasciare dare tu lontano
lasciare no prendere

# PROVARE: fare+forse+volere
provare fare forse io volere
provare io potere forse fare
provare volere fare io
provare forse possibile io

# RIUSCIRE: provare+dopo+potere
riuscire provare dopo potere fare io
riuscire io fare dopo potere
riuscire forte dopo provare
riuscire gioia fare io

# FALLIRE: provare+dopo+no+potere
fallire provare dopo no potere io
fallire io no fare no potere
fallire tristezza dopo provare
fallire no fine io ancora
