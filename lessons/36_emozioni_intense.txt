# Lezione 36: Emozioni Intense (EMOZIONE)
# Dominio: stati emotivi estremi, picchi affettivi
# Ancore da: lezioni 00-35 (846 parole note)

# === GIOIA INTENSA ===
euforia: gioia forte grande io dentro / tristezza
estasi: gioia forte grande io dentro / dolore
entusiasmo: gioia forte io dentro fare / calma
eccitazione: gioia forte veloce io dentro / calma
esaltazione: gioia forte grande io alto / basso
esultanza: gioia forte grande io fuori / tristezza

# === TRISTEZZA PROFONDA ===
mestizia: tristezza forte dentro io lento / gioia
afflizione: tristezza forte dentro io dolore / gioia
desolazione: tristezza forte dentro io solo / insieme
disperazione: tristezza forte dentro io no / speranza
sconforto: tristezza forte dentro io debole / forte

# === SERENITÀ ===
serenita: calma dentro io forte luce / paura
tranquillita: calma dentro io lento forte / veloce
pace: calma dentro io forte insieme / paura
quiete: calma dentro io lento piccolo / forte
placidita: calma dentro io lento forte / veloce

# === AGITAZIONE ===
agitazione: paura dentro io veloce forte / calma
trepidazione: paura dentro io veloce dopo / calma
turbamento: paura dentro io forte dentro / calma
smarrimento: paura dentro io no sapere / sapere
