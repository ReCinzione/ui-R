# Lezione 43: Memoria e Ricordi (MEMORIA)
# Dominio: processi mnemonici, passato
# Ancore da: lezioni 00-42 (939+ parole note)

# === RICORDARE ===
rammentare: ricordare io dentro prima pensare / dimenticare
rimembrare: ricordare io dentro prima forte / dimenticare
revocare: ricordare io dentro prima forte / dimenticare
riesumare: ricordare io dentro prima lontano / ora
reminiscenza: ricordare io dentro prima dentro / dimenticare

# === DIMENTICARE ===
scordare: dimenticare io dentro no ricordare / ricordare
cancellare: dimenticare io dentro forte no / ricordare
rimuovere: dimenticare io dentro forte no / ricordare
seppellire: dimenticare io dentro dentro no / fuori

# === CONSERVARE ===
custodire: ricordare io dentro forte sempre / perdere
preservare: ricordare io dentro forte sempre / perdere
trattenere: ricordare io dentro forte dentro / lasciare
serbare: ricordare io dentro forte sempre / perdere

# === EVOCARE ===
evocare: ricordare io dentro forte fare / no
suscitare: ricordare io dentro forte fare / no
risvegliare: ricordare io dentro forte nuovo / dormire

# === IMPRESSIONE ===
imprimere: ricordare io dentro forte sempre / no
incidere: ricordare io dentro forte dentro / fuori
scolpire: ricordare io dentro forte sempre / no
marcare: ricordare io dentro forte dentro / no

# === OBLIO ===
obliare: dimenticare io dentro lento no / ricordare
svanire: dimenticare io dentro lento lontano / qui
dissolversi: dimenticare io dentro lento no / forte
sbiadire: dimenticare io dentro lento debole / forte
